143 using VarT = typename VarVecT::value_type;
144 VarT val = 0.;
145
146 // assert tau does not belong to last level
147 unsigned var_idx = this->var_map(tau);
148 for (unsigned i = 0; i < feats.n_elem; i++){ // perform dot product
149 val += feats(i) * (vars[var_idx++]); // cannot use iterators with pragma (dependent)

151 // normalise by number of features
